
        /*fibonaci
            iN MC

        */
        void main()
            {
                int i, n, t1, t2, next;
                t1 = 0;
                t2 = 0;
                print("the number of elements: ");
                input(n);
                print("Fibonacci: ");
                for (i = 1; i <= n; i = i + 1)
                {
                    print(t1, ", ");
                    next = t1 + t2;
                    t1 = t2;
                    t2 = next;
                }
                return;
            }
